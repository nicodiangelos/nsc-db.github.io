CREATE TABLE unit(
    charid int,
    cardid int,
    charaid int,
    abilityid1 int,
    abilityid2 int,
    skillid1 int,
    skillid2 int,
    leadid int,
    jpname varchar(255) COLLATE ujis_japanese_ci,
    engname varchar(255),
    attribute int,
    affi varchar(255),
    cost int,
    hp int,
    atk int,
    def int,
    spd int,
    er int,
    cd int,
    cc int,
    rating int
);

CREATE TABLE name(
	charaid int,
	jpname varchar(255) COLLATE ujis_japanese_ci,
	engname varchar(255),
	detail varchar(255)
);
CREATE TABLE skill(
	skillid int,
    nameid int,
    typeid int,
    castspeed int,
    jpdesc varchar(255) COLLATE ujis_japanese_ci,
    engdesc varchar(255),
    cost int
);

CREATE TABLE skillName(
  	nameid int,
    jpname varchar(255) COLLATE ujis_japanese_ci,
    engname varchar(255)
);

CREATE TABLE ability(
	abilityid int,
   	jpname varchar(255) COLLATE ujis_japanese_ci,
    engname varchar(255),
    jpdesc varchar(255) COLLATE ujis_japanese_ci,
    engdesc varchar(255)
);

CREATE TABLE lead(
    leadid int,
    jpname varchar(255) COLLATE ujis_japanese_ci,
    engname varchar(255),
    jpdesc varchar(255) COLLATE ujis_japanese_ci,
    engdesc varchar(255)
);

CREATE TABLE gear(
	gearid int,
    geartype int,
    jpname varchar(255) COLLATE ujis_japanese_ci,
    engname varchar(255),
    jpnorm varchar(255) COLLATE ujis_japanese_ci,
    engnorm varchar(255),
    jpuni varchar(255) COLLATE ujis_japanese_ci,
    enguni varchar(255),
    chara varchar(255)
);
